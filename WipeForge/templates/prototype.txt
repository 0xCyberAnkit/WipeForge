    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const hamburgerButton = document.getElementById('hamburger-button');
            const closeButton = document.getElementById('close-button');
            const mobileMenu = document.getElementById('mobile-menu');

            hamburgerButton.addEventListener('click', () => {
                mobileMenu.classList.add('open');
            });

            closeButton.addEventListener('click', () => {
                mobileMenu.classList.remove('open');
            });

            // Prototype functionality
            const wipeForm = document.getElementById('wipe-form');
            const formSection = document.getElementById('form-section');
            const statusSection = document.getElementById('status-section');
            const resultsSection = document.getElementById('results-section');
            const progressBar = document.getElementById('progress-bar');
            const statusMessage = document.getElementById('status-message');
            const completedDeviceName = document.getElementById('completed-device-name');
            const completedDeviceId = document.getElementById('completed-device-id');
            const completedMethod = document.getElementById('completed-method');
            const downloadLog = document.getElementById('download-log');

            wipeForm.addEventListener('submit', (e) => {
                e.preventDefault();

                const deviceName = document.getElementById('deviceName').value;
                const deviceId = document.getElementById('deviceId').value;
                const wipeMethod = document.getElementById('wipeMethod').value;
                
                // Hide the form and show the status section
                formSection.classList.add('hidden');
                statusSection.classList.remove('hidden');
                resultsSection.classList.add('hidden');
                
                // Reset progress bar
                progressBar.style.width = '0%';
                
                const steps = [
                    { message: 'Validating device information...', duration: 1000 },
                    { message: `Starting ${wipeMethod} sanitization on ${deviceName} (${deviceId})...`, duration: 2000 },
                    { message: 'Overwriting data blocks with zero-pass...', duration: 3000 },
                    { message: 'Verifying wipe integrity...', duration: 2500 },
                    { message: 'Generating tamper-proof log...', duration: 1500 }
                ];

                let currentProgress = 0;
                let stepIndex = 0;

                const runWipe = () => {
                    if (stepIndex < steps.length) {
                        const step = steps[stepIndex];
                        statusMessage.textContent = step.message;
                        currentProgress += 20; // Each step is 20%
                        progressBar.style.width = `${currentProgress}%`;

                        setTimeout(() => {
                            stepIndex++;
                            runWipe();
                        }, step.duration);
                    } else {
                        // All steps complete
                        setTimeout(() => {
                            statusSection.classList.add('hidden');
                            resultsSection.classList.remove('hidden');
                            completedDeviceName.textContent = deviceName;
                            completedDeviceId.textContent = deviceId;
                            completedMethod.textContent = wipeMethod;
                        }, 500);
                    }
                };

                runWipe();
            });
            
            // Dummy download functionality
            downloadLog.addEventListener('click', () => {
                const logContent = `
WipeForge Data Sanitization Log
-------------------------------
Device Name: ${document.getElementById('deviceName').value}
Device ID: ${document.getElementById('deviceId').value}
Wiping Method: ${document.getElementById('wipeMethod').value}
Status: COMPLETE
Date: ${new Date().toISOString()}
Log: All data blocks successfully overwritten and verified.
`;
                const blob = new Blob([logContent], { type: 'text/plain' });
                const a = document.createElement('a');
                a.href = URL.createObjectURL(blob);
                a.download = `wipe_log_${document.getElementById('deviceId').value}.txt`;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
            });
        });
    </script>
